    ```cpp

    {
        int a = 5;

        std::cout << "b: " << b << '\n';

        return 0;
    }
    ```

- #### II.
    ```cpp
    #include <iostream>

    int main()
    {
        int a = 5;

        std::cout << "a: " << a << '\n';

        int b = a;
        a = 7;

        std::cout << "a: " << a << '\n';
        std::cout << "b: " << b << '\n';

        return 0;
    }
    ```

- #### III.
    ```cpp
    #include <iostream>

    int main()
    {
        int a = 5;

        std::cout << "a: " << a << '\n';

        int b = 2 * a;

        std::cout << "a: " << a << '\n';
        std::cout << "b: " << b << '\n';

        int a = 2 * b;

        std::cout << "a: " << a << '\n';
        std::cout << "b: " << b << '\n';

        return 0;
    }
    ```

- #### IV.
