    ```py
    ages['rani'] += 1
    print(ages['rani'])

    def incrementAge(name):
    ```

- #### II.
    ```py
    prices = {'pen': 10, 'pencil': 5, 'eraser': 3, 'sharpener': 8, 'notebook': 25, 'pen': 12}

    print(stock)
    ```

- #### III.
    ```py
    ages = {'hema': 23, 'rekha': 30, 'jaya': 27, 'sushma': 39}

    print(ages.get('rani'))
    print(ages['alia'])

    removed_item_1 = ages.pop('rekha')
    print(removed_item_1)

    def printDict():
        print("### People's Ages ###")
        for name in ages:
            print(str(name) + ": " + str(ages[name]))

    printDict()

    removed_item_2 = ages.pop('mahi')
    print(removed_item_2)

    del ages['maya']
    removed_item_3 = del ages['jaya']
    print(removed_item_3)

    printDict()
    ```

- #### IV.
    ```py
    fruit_stock = {'apple': 10, 'banana': 18, 'orange': 15, 'papaya': 12}

    print(f'Total fruits: {len(fruit_stock)}')

    fruit_stock.clear()
    ```
