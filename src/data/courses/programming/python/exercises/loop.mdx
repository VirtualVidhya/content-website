    ```py
    print("\nEnd of Loop")
    ```

- #### IV.
    ```py
    count = 1

    while count <= 10:
        print(count, end=' ')

        if count == 3:
            break

        count += 1
        
    print("\nEnd of Loop")
    ```

- #### V. **_(Adv.)_**
    ```py
    i = 50

    while (i > 0 or (i + i == i * i)):

    print("\nEnd of Loop")
    ```

- #### VI.
    ```py
    i = -5

    while (i <= 0):
    ```
